// Given an integer x, return true if x is a 
// palindrome, and false otherwise.

class Solution {
public:
    bool isPalindrome(int x) {
        //One way by converting to string. But takes more time though in order of n.
        // string temp = to_string(x);
        // int size= temp.size();
        // for(int i=0;i<size;i++){
        //     if(temp[i]!=temp[size-i-1]){
        //         return false;
        //     }
        // }
        // cout<<size;
        // return true;
        
        int p=x;
        int digitCount=0;
        while(p){
            p/=10;
            digitCount++;
        }
        //Returns fals for all negetive numbers and multiple of 10 
        if(x<0 || (x%10==0 && x!=0)) return false;

        int num=x;
        int revnum=0;

        while(num>revnum){
            int ldigit= num%10;
            revnum= 10*revnum + ldigit;
            num/=10;
        }
        //if digit count is odd then ignore the middle digit
        if(digitCount%2!=0)
            revnum/=10;
        
        if(revnum==num )
            return true;
        else return false;

    }
};
